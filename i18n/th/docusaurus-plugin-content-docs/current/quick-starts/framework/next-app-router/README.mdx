### ดึงโทเค็นการเข้าถึงสำหรับทรัพยากร API \{#fetch-access-token-for-the-api-resource}

<FetchAccessTokenForApiResources />

หากคุณต้องการดึงโทเค็นการเข้าถึงใน server component สามารถใช้ฟังก์ชัน `getAccessTokenRSC` ได้ดังนี้:

```tsx title="app/page.tsx"
// ตัวอย่างนี้เป็นโค้ด TypeScript สำหรับดึง access token ใน server component
import { getAccessTokenRSC } from '@logto/next/server-actions';
import { logtoConfig } from './logto';

export default async function Home() {
  const accessToken = await getAccessTokenRSC(logtoConfig, 'https://shopping.your-app.com/api');

  return (
    <main>
      <p>Access token: {accessToken}</p>
    </main>
  );
}
```

:::tip

HTTP ไม่อนุญาตให้ตั้งค่า cookie หลังจากเริ่ม streaming แล้ว ดังนั้น `getAccessTokenRSC` จะไม่สามารถอัปเดตค่า cookie ได้ หาก access token ถูกรีเฟรช จะไม่ถูกบันทึกใน session แนะนำให้ใช้ฟังก์ชัน `getAccessToken` ในฝั่ง client หรือ route handlers แทน

:::

### ดึงโทเค็นองค์กร (Organization tokens) \{#fetch-organization-tokens}

<FetchOrganizationTokenForUser />

หากคุณต้องการดึงโทเค็นองค์กรใน server component สามารถใช้ฟังก์ชัน `getOrganizationTokenRSC` ได้ดังนี้:

```tsx title="app/page.tsx"
// ตัวอย่างนี้เป็นโค้ด TypeScript สำหรับดึง organization token ใน server component
import { getOrganizationTokenRSC } from '@logto/next/server-actions';
import { logtoConfig } from './logto';

export default async function Home() {
  const token = await getOrganizationTokenRSC(logtoConfig, 'organization-id');

  return (
    <main>
      <p>Organization token: {token}</p>
    </main>
  );
}
```

:::tip

HTTP ไม่อนุญาตให้ตั้งค่า cookie หลังจากเริ่ม streaming แล้ว ดังนั้น `getOrganizationTokenRSC` จะไม่สามารถอัปเดตค่า cookie ได้ หาก access token ถูกรีเฟรช จะไม่ถูกบันทึกใน session แนะนำให้ใช้ฟังก์ชัน `getOrganizationToken` ในฝั่ง client หรือ route handlers แทน

## การ redirect ไปยังหน้าลงชื่อเข้าใช้อัตโนมัติเมื่อไม่ได้รับอนุญาต \{#automatic-sign-in-redirect\}

:::tip
ตัวช่วย `signIn` จะปรับเปลี่ยน cookie เพื่อสร้าง session การลงชื่อเข้าใช้ จึงไม่สามารถเรียกใช้โดยตรงใน React Server Component (RSC) ได้ หากต้องการให้ระบบลงชื่อเข้าใช้อัตโนมัติเมื่อ RSC ตรวจพบผู้ใช้ที่ไม่ได้รับอนุญาต ให้เรียก `signIn` ใน route handler เฉพาะ และ redirect ไปยัง route นั้น
:::

```ts title="app/sign-in/route.ts"
import { signIn } from '@logto/next/server-actions';
import { logtoConfig } from '../../logto';

export async function GET() {
  await signIn(logtoConfig);
}
```

```tsx title="app/protected/page.tsx"
import { getLogtoContext } from '@logto/next/server-actions';
import { redirect } from 'next/navigation';
import { logtoConfig } from '../logto';

export default async function ProtectedPage() {
  const { isAuthenticated } = await getLogtoContext(logtoConfig);

  if (!isAuthenticated) {
    redirect('/sign-in');
  }

  return <div>Protected content</div>;
}
```
