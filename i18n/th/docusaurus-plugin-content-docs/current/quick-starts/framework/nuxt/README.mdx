## ใช้งานใน middleware หรือ API routes \{#use-in-middleware-or-api-routes}

Composable `useLogtoClient()` และ `useLogtoUser()` ไม่สามารถใช้ใน middleware หรือ API routes ได้ คุณสามารถใช้ฟังก์ชัน `logtoEventHandler()` เพื่อรับ Logto client และ context อื่น ๆ ได้ดังนี้:

```ts title="/server/api/auth/get-access-token.ts"
import { logtoEventHandler } from '#logto';

// ดึง accessToken ของผู้ใช้ที่เข้าสู่ระบบ
export default defineEventHandler(async (event) => {
  const config = useRuntimeConfig(event);
  await logtoEventHandler(event, config);
  const accessToken = await event.context.logtoClient.getAccessToken();
  return { accessToken };
});
```
