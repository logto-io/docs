{/* ส่วนนี้รวมการตั้งค่า redirect URI และ post sign-out redirect URI ซึ่งอาจเป็นประโยชน์สำหรับเฟรมเวิร์กบางประเภท */}

export const defaultBaseUrl = 'http://localhost:3000/';
export const defaultRedirectUri = `${defaultBaseUrl}callback`;

#### กำหนดค่า Redirect URI \{#configure-redirect-uris}

ไปที่หน้ารายละเอียดแอปพลิเคชันใน Logto Console เพิ่ม redirect URI <code>{props.redirectUri ?? 'http://localhost:3000/callback'}</code>

<img alt="Redirect URI in Logto Console" src="/img/assets/web-redirect-uri.png" width="600px" />

เช่นเดียวกับการลงชื่อเข้าใช้ ผู้ใช้ควรถูกเปลี่ยนเส้นทางไปที่ Logto เพื่อออกจากเซสชันที่ใช้ร่วมกัน เมื่อเสร็จสิ้นแล้ว ควรเปลี่ยนเส้นทางผู้ใช้กลับไปยังเว็บไซต์ของคุณ ตัวอย่างเช่น เพิ่ม <code>{defaultBaseUrl}</code> ในส่วน post sign-out redirect URI

จากนั้นคลิก "Save" เพื่อบันทึกการเปลี่ยนแปลง
