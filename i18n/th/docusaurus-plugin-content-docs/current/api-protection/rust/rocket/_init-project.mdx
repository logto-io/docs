ในการเริ่มต้นโปรเจกต์ Rocket ใหม่ ให้สร้างไดเรกทอรีและตั้งค่าโครงสร้างพื้นฐานดังนี้:

```bash
cargo new your-api-name
cd your-api-name
```

เพิ่ม dependencies ของ Rocket ลงใน `Cargo.toml` ของคุณ:

```toml title="Cargo.toml"
[dependencies]
rocket = { version = "0.5", features = ["json"] }
serde = { version = "1.0", features = ["derive"] }
serde_json = "1.0"
```

สร้างแอปพลิเคชัน Rocket พื้นฐาน:

```rust title="src/main.rs"
// นำเข้าไลบรารีที่จำเป็น
use rocket::{get, launch, routes, serde::json::Json};
use serde_json::{json, Value};

// สร้าง handler สำหรับเส้นทางหลัก
#[get("/")]
fn hello_handler() -> Json<Value> {
    Json(json!({ "message": "Hello from Rocket" }))
}

// ฟังก์ชันเริ่มต้น Rocket
#[launch]
fn rocket() -> _ {
    rocket::build()
        .mount("/", routes![hello_handler])
}
```

เริ่มต้นเซิร์ฟเวอร์สำหรับพัฒนา:

```bash
cargo run
```

:::note
ดูรายละเอียดเพิ่มเติมเกี่ยวกับการตั้งค่าเส้นทาง (routes), ตัวป้องกันคำขอ (request guards) และฟีเจอร์อื่น ๆ ได้จากเอกสาร Rocket
:::
