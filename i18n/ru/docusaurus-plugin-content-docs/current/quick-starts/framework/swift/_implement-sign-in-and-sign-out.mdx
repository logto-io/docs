#### Вход и выход \{#sign-in-and-sign-out}

<SignInNote calling=".signInWithBrowser(redirectUri:)" />

Вы можете использовать `client.signInWithBrowser(redirectUri:)` для входа пользователя и `client.signOut()` для выхода пользователя.

Например, в приложении SwiftUI:

```swift title="ContentView.swift"
struct ContentView: View {
  @State var isAuthenticated: Bool

  init() {
    isAuthenticated = client.isAuthenticated
  }

  var body: some View {
    VStack {
      if isAuthenticated {
        Button("Sign Out") {
          Task { [self] in
            await client.signOut()
            isAuthenticated = false
          }
        }
      } else {
        Button("Sign In") {
          Task { [self] in
            do {
              try await client.signInWithBrowser(redirectUri: "${
                props.redirectUris[0] ?? 'io.logto://callback'
              }")
              isAuthenticated = true
            } catch let error as LogtoClientErrors.SignIn {
              // ошибка произошла во время входа
            } catch {
              // другие ошибки
            }
          }
        }
      }
    }
  }
}
```
