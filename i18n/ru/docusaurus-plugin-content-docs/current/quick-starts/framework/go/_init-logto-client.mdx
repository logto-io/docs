Сначала создайте конфигурацию Logto:

```go title="main.go"
func main() {
	// ...
	logtoConfig := &client.LogtoConfig{
		Endpoint:  "<your-logto-endpoint>", // Например, http://localhost:3001
		AppId:     "<your-application-id>",
		AppSecret: "<your-application-secret>",
	}
	// ...
}
```

Затем вы можете создать `LogtoClient` для каждого пользовательского запроса с использованием вышеуказанной конфигурации Logto:

```go title="main.go"
func main() {
	// ...

	router.GET("/", func(ctx *gin.Context) {
		// Создание LogtoClient
		session := sessions.Default(ctx)
		logtoClient := client.NewLogtoClient(
			logtoConfig,
			&SessionStorage{session: session},
		)

		// Используйте Logto для управления содержимым главной страницы
		authState := "Вы не вошли на этот сайт. :("

		if logtoClient.IsAuthenticated() {
			authState = "Вы вошли на этот сайт! :)"
		}

		homePage := `<h1>Hello Logto</h1>` +
			"<div>" + authState + "</div>"

		ctx.Data(http.StatusOK, "text/html; charset=utf-8", []byte(homePage))
	})

	// ...
}
```
