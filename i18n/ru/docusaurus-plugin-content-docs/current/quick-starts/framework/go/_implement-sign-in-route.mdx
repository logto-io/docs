После настройки перенаправления URI, мы добавляем маршрут `sign-in` для обработки запроса на вход и также добавляем ссылку для входа на главной странице:

```go title="main.go"
func main() {
	// ...

	// Добавляем ссылку для выполнения запроса на вход на главной странице
	router.GET("/", func(ctx *gin.Context) {
		// ...
		homePage := `<h1>Hello Logto</h1>` +
			"<div>" + authState + "</div>" +
			// Добавляем ссылку
			`<div><a href="/sign-in">Sign In</a></div>`

		ctx.Data(http.StatusOK, "text/html; charset=utf-8", []byte(homePage))
	})

	// Добавляем маршрут для обработки запросов на вход
	router.GET("/sign-in", func(ctx *gin.Context) {
		session := sessions.Default(ctx)
		logtoClient := client.NewLogtoClient(
			logtoConfig,
			&SessionStorage{session: session},
		)

		// Запрос на вход обрабатывается Logto.
		// Пользователь будет перенаправлен на URI перенаправления после входа.
		signInUri, err := logtoClient.SignIn("http://localhost:3000/callback")
		if err != nil {
			ctx.String(http.StatusInternalServerError, err.Error())
			return
		}

		// Перенаправляем пользователя на страницу входа Logto.
		ctx.Redirect(http.StatusTemporaryRedirect, signInUri)
	})

	// ...
}
```

Теперь, когда ваш пользователь посетит `http://localhost:3000/sign-in`, он будет перенаправлен на страницу входа Logto.
