Сначала создадим страницу "Callback" в Webflow и просто разместим на ней статический текст "Перенаправление...". Затем добавьте следующий пользовательский код на уровне страницы на страницу "Callback".

```html
<script type="module">
  (async () => {
    // Обработка логики обратного вызова входа с помощью вызова метода SDK
    await logtoClient.handleSignInCallback(window.location.href);

    // Перенаправление обратно на главную страницу после завершения обработки
    window.location.assign('https://your-awesome-site.webflow.io');
  })();
</script>
```
