#### Implement a sign-out button

```ts
import { useLogto } from '@logto/vue';

const { signOut } = useLogto();
const onClickSignOut = () => signOut('http://localhost:3000');
```

```html
<button @click="onClickSignOut">Sign Out</button>
```
